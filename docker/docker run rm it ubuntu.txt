docker run --rm -it ubuntu bash -c "
  # 1. Set Timezone to Asia/Seoul
  ln -snf /usr/share/zoneinfo/Asia/Seoul /etc/localtime && echo 'Asia/Seoul' > /etc/timezone && \
  
  # 2. Update Sources to Kakao Mirror
  sed -i 's|http://archive.ubuntu.com/ubuntu/|http://mirror.kakao.com/ubuntu|g' /etc/apt/sources.list.d/ubuntu.sources && \
  sed -i 's|http://security.ubuntu.com/ubuntu/|http://mirror.kakao.com/ubuntu|g' /etc/apt/sources.list.d/ubuntu.sources && \
  
  # 3. Install packages and setup user
  apt-get update && apt-get install -y sudo nano git curl && \
  curl -L -o /tmp/fastfetch.deb https://github.com/fastfetch-cli/fastfetch/releases/latest/download/fastfetch-linux-amd64.deb && \
  apt-get install -y /tmp/fastfetch.deb && rm /tmp/fastfetch.deb && \
  
  useradd -m -G sudo -s /bin/bash tempuser && \
  echo 'tempuser:helloworld' | chpasswd && \
  runuser -u tempuser -- bash -c 'cd; fastfetch; exec bash'"

# fedora login + fastfetch  
docker run --rm -it fedora bash -c "
  # 1. System Setup (as root)
  ln -snf /usr/share/zoneinfo/Asia/Seoul /etc/localtime && \
  echo 'max_parallel_downloads=20' >> /etc/dnf/dnf.conf && \
  dnf install -y sudo nano git curl fastfetch && \
  
  # 2. User Setup
  useradd -m -G wheel -s /bin/bash tempuser && \
  echo 'tempuser:helloworld' | chpasswd && \
  echo 'tempuser ALL=(ALL) NOPASSWD: ALL' >> /etc/sudoers && \

  # 3. Use runuser to start a clean interactive login shell
  runuser -u tempuser -- bash -c 'cd; fastfetch; exec bash'"

docker run --rm -it archlinux bash -c "
  # 1. Set Timezone
  ln -snf /usr/share/zoneinfo/Asia/Seoul /etc/localtime && \

  # 2. Setup Speed: Enable Parallel Downloads (5 is standard for Arch)
  sed -i 's/^#ParallelDownloads = 5/ParallelDownloads = 5/' /etc/pacman.conf && \ 

  # 4. Install packages and setup user
  pacman -S --noconfirm sudo nano git curl fastfetch && \
  useradd -m -G wheel -s /bin/bash tempuser && \
  echo 'tempuser:helloworld' | chpasswd && \
  echo '%wheel ALL=(ALL:ALL) ALL' > /etc/sudoers.d/wheel && \
  runuser -u tempuser -- bash -c 'cd; fastfetch; exec bash'"
